mydata <- read.csv("data/tedsd_puf_2017.csv")
medexp <- read.csv("data/medicaid_expansion.csv")
# merge
mydata <- merge(mydata, medexp, by='STFIPS')

response = "COMPLETED"

teds[, response] = mydata[, response]
